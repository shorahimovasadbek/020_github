1. Win+R => Run buyruq (cmd) enter buyrug'i yoziladi
2. git versiyasini ko'rish => git --version
3. gitda o'zini tanishtirish yoki yanig user yaratish => git config --global user.name "username"
4. gitda o'zini tanishtirish yoki yanig user yaratish email => git config --global user.email "useremail"
5. gitni ishga tushurish => git init
6. ls => biz turgan papkadagi filelar va folderlarni ko'rsatadi
7. ls -all => biz turgan papkadagi filelar va folderlarni hammasini ko'rsatadi.
8. git status => hozirgi jarayonni ko'rsatadi
9. git add (file name) => bu yani yaratilgan fileni git ga qo'shib beradi.
10. git add . => barchasini qo'shib qo'yadi.
11. git commit -m "commit" 
12. git diff => barcha o'zgarishlarni ko'rish
13. git log => vaqtigacha ko'rsatadi.
14. mkdir (folder name) => folder yaratish
15. rmdir (folder name) => folder o'chirish
16. cat > (file name) => file yaratish
17. rm (file name) => file o'chirish
18. git remote add origin https://github.com/shorahimovasadbek/020_github.git => push qilish joyini ko'rsatish
19. git push -u origin master => master branchiga qo'shish 
20. git clone https://github.com/shorahimovasadbek/020_github.git => repositorydan olib kelish projectni
